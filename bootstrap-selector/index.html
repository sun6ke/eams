Doc: Bootstrap Select

给selector添加.selectpicker启用bootstrap-select 
<select class="form-control selectpicker">
  <option value="">...</option>
  <option value="1">Option A</option>
  <option value="2">Option B</option>
  <option value="3">Option C</option>
</select>
<select class="form-control selectpicker" name="multiple-select" id="multiple-select" multiple>
  <option value="1">Option A</option>
  <option value="2">Option B</option>
  <option value="3">Option C</option>
</select>
添加data-style="btn-primary" 
<select class="form-control selectpicker" data-style="btn-primary">
  <option value="">...</option>
  <option value="1">Option A</option>
  <option value="2">Option B</option>
  <option value="3">Option C</option>
</select>
添加data-max-options="2" 
<select class="form-control selectpicker" data-max-options="2" multiple>
  <option value="1">Option A</option>
  <option value="2">Option B</option>
  <option value="3">Option C</option>
</select>
